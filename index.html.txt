        // Crear partículas flotantes
        function createParticles() {
            const particlesContainer = document.getElementById('particles');
            const particleCount = 80;
            
            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                
                // Posición inicial aleatoria
                particle.style.left = Math.random() * 100 + '%';
                
                // Duración de animación aleatoria
                const duration = 15 + Math.random() * 20;
                particle.style.animationDuration = duration + 's';
                
                // Retraso aleatorio
                particle.style.animationDelay = Math.random() * 10 + 's';
                
                // Deriva horizontal aleatoria
                const drift = (Math.random() - 0.5) * 200;
                particle.style.setProperty('--drift', drift + 'px');
                
                // Tamaño aleatorio
                const size = 2 + Math.random() * 4;
                particle.style.width = size + 'px';
                particle.style.height = size + 'px';
                
                particlesContainer.appendChild(particle);
            }
        }

        // Crear partículas adicionales en los banners
        function createBannerParticles() {
            const banners = document.querySelectorAll('.banner-section');
            
            banners.forEach(banner => {
                for (let i = 0; i < 15; i++) {
                    const particle = document.createElement('div');
                    particle.style.position = 'absolute';
                    particle.style.width = (2 + Math.random() * 3) + 'px';
                    particle.style.height = particle.style.width;
                    particle.style.background = `radial-gradient(circle, rgba(13, 110, 253, ${0.4 + Math.random() * 0.4}), rgba(110, 168, 254, ${0.2 + Math.random() * 0.3}))`;
                    particle.style.borderRadius = '50%';
                    particle.style.left = Math.random() * 100 + '%';
                    particle.style.top = Math.random() * 100 + '%';
                    particle.style.pointerEvents = 'none';
                    particle.style.boxShadow = '0 0 10px rgba(13, 110, 253, 0.6)';
                    
                    const duration = 3 + Math.random() * 4;
                    const delay = Math.random() * 2;
                    const translateX = (Math.random() - 0.5) * 50;
                    const translateY = (Math.random() - 0.5) * 50;
                    
                    particle.style.animation = `bannerParticleFloat ${duration}s ${delay}s ease-in-out infinite`;
                    particle.style.setProperty('--tx', translateX + 'px');
                    particle.style.setProperty('--ty', translateY + 'px');
                    
                    banner.appendChild(particle);
                }
            });
            
            // Agregar animación para las partículas del banner
            const style = document.createElement('style');
            style.textContent = `
                @keyframes bannerParticleFloat {
                    0%, 100% {
                        transform: translate(0, 0) scale(1);
                        opacity: 0.6;
                    }
                    50% {
                        transform: translate(var(--tx), var(--ty)) scale(1.5);
                        opacity: 1;
                    }
                }
            `;
            document.head.appendChild(style);
        }

        // Función para expandir/contraer períodos
        function togglePeriod(event, card) {
            // Evitar que se cierre si se hace clic en las secciones de contenido
            if (event.target.closest('.content-section')) {
                return;
            }
            
            const isExpanded = card.classList.contains('expanded');
            
            // Cerrar todas las tarjetas
            document.querySelectorAll('.period-card').forEach(c => {
                c.classList.remove('expanded');
            });
            
            // Abrir la seleccionada si no estaba expandida
            if (!isExpanded) {
                card.classList.add('expanded');
            }
        }

        // Inicializar
        document.addEventListener('DOMContentLoaded', function() {
            // Crear partículas de fondo
            createParticles();
            
            // Crear partículas en los banners
            createBannerParticles();
            
            // Prevenir que los clics en las secciones de contenido cierren la tarjeta
            document.querySelectorAll('.content-section').forEach(section => {
                section.addEventListener('click', function(e) {
                    e.stopPropagation();
                    console.log('Sección clickeada:', this.querySelector('.content-title').textContent);
                });
            });
        });
